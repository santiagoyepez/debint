Servicio Carrera

	Operación			:	Registrar carrera
	Método				:	POST
	URL					:	[servidor]/[proyecto]/utl/carrera/registrar
	Parámetro esperado	:	Carrera (JSON)
	Ejemplo parámetro	:	{"carrera":"mis bolas"}
	
	Operación			:	Listar todo
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/carrera/listar
	Parámetro esperado	:	ninguno

	Operación			:	Consultar por idCarrera
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/carrera/consultar/{idCarrera}
	Parámetro esperado	:	idCarrera (entero)

Servicio Recurso

	Operación			:	Registrar recurso
	Método				:	POST
	URL					:	[servidor]/[proyecto]/utl/recurso/registrar
	Parámetro esperado	:	Recurso (JSON)
	Ejemplo parámetro	:	{usuario:{"idUsuario":1},categoria:{"idCategoria":1},"titulo":"mis bolas","descripcion":"garganta profunda","visible":1,"solicitudRecurso":{"idSolicitud":1}}
	Nota				:	El campo "visible" debe estar en {0, 1}, no otros valores. El campo "solicitudRecurso" debe ser igual a null si no hay solicitud.
	
	Operación			:	Listar todo
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/recurso/listar
	Parámetro esperado	:	ninguno

	Operación			:	Consultar por idRecurso
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/recurso/consultar/{idRecurso}
	Parámetro esperado	:	idRecurso (entero)
	
	Operación			:	Consultar por categoría
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/recurso/consultarPorCategoria/{idCategoria}
	Parámetro esperado	:	idCategoria (entero)
	
	Operación			:	Consultar por título
	Método				:	GET
	URL					:	[servidor]/[proyecto]/utl/recurso/consultarPorTitulo/{titulo}
	Parámetro esperado	:	titulo (string)
